def remove_review_special(text):
    
    # menghilangkan tab, new line, dan back slice
    text = text.replace('\\t'," ").replace('\\n'," ").replace('\\u'," ").replace('\\',"")
    
    # menghilangkan non ASCII (emoticon, chinese word, dll)
    text = text.encode('ascii', 'ignore').decode('ascii') 
    
    # menghilangkan mention, link, hashtag
    text = ' '.join(re.sub("([@#][A-Za-z0-9]+)|(\w+:\/\/\S+)"," ", text).split())
    
    return text

def remove_number(text):
    return  re.sub(r"\d+", "", text)


def remove_punctuation(text):
    return text.translate(str.maketrans("","",string.punctuation))


# menghilangkan whitespace leading & trailing
def remove_whitespace_LT(text):
    return text.strip()


# mengubah spasi yang berlebihan menjadi spasi tunggal
def remove_whitespace_multiple(text):
    return re.sub('\s+',' ',text)


def remove_single_char(text):
    return re.sub(r"\b[a-zA-Z]\b", "", text)
